\[
\rho = \left(
\begin{array}{cc}
|\alpha|^2 & \alpha \beta^{\ast} e^{-i\omega_0 t} C(t) \\
\alpha^{\ast} \beta e^{i\omega_0 t} C(t) & |\beta|^2
\end{array}
\right)
\]
